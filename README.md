# crawl_project
爬虫项目仓库

settings.py 设置数据库连接,例如:
MYSQL_HOST='127.0.0.1'
MYSQL_DB='your database'
MYSQL_USER='your user'
MYSQL_PASSWD='your password'


constant.py 设置用户密码
#知乎的用户密码
ZHIHU_USER = "你的用户名"      #用户名
ZHIHU_PSWD = "你的密码"   #密码

#天天视图账号密码http://www.ttshitu.com/  你也可以使用别的接口
TTST_USER = "天天视图用户名"     #用户名
TTST_PSWD = '天天视图密码'     #密码


运行main.py
